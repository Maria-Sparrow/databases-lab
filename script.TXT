FROM classes c
WHERE UPPER(c.country) = 'UKRAINE' AND EXISTS (
SELECT c.country, c.class
FROM classes c
WHERE UPPER(c.country) = 'UKRAINE' )

SELECT c.country, c.class
FROM classes c
WHERE NOT EXISTS (SELECT c.country, c.class
FROM classes c
WHERE UPPER(c.country) = 'UKRAINE' )

7)USE ships_db;

FROM ships

9)use firm_sql;
select outcome2.point, outcome2.date,
case when sum(outcome1.out) > outcome2.out then 'more than once a day'
when sum(outcome1.out) < outcome2.out or sum(outcome1.out) is null then 'once a day'
when sum(outcome1.out) = outcome2.out then 'both'
end as text
from outcome outcome1 LEFT JOIN outcome_o outcome2
on outcome1.date = outcome2.date
group by outcome1.date
union
select outcome2.point, outcome2.date,
case when sum(outcome1.out) > outcome2.out then 'more than once a day'
when sum(outcome1.out) < outcome2.out or sum(outcome1.out) is null then 'once a day'
when sum(outcome1.out) = outcome2.out then 'both'
end as text
from outcome outcome1 RIGHT JOIN outcome_o outcome2
on outcome1.date = outcome2.date
group by outcome1.date
